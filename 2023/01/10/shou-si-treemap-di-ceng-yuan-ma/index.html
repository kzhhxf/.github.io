public class HashMap<K,V> extends AbstractMap<K,V> implements Map<K,V>{
    
    //初始化容量（必须是2的幂）
    static final int DEFAULT_INITIAL_CAPACITY = 1 << 4; // aka 16
    //最大容量
    static final int MAXIMUM_CAPACITY = 1 << 30;
    //默认的负载因子
    static final float DEFAULT_LOAD_FACTOR = 0.75f;
    //长度为0的数组
    static final Entry<?,?>[] EMPTY_TABLE = {};
    //hash表/hash数组
    transient Entry<K,V>[] table = (Entry<K,V>[]) EMPTY_TABLE;//new Entry[16];
    //映射关系的个数
    transient int size;//0
    //阈值（数组长度*负载因子）
    int threshold;//12
    //负载因子
    final float loadFactor;//0.75f
    //操作数（记录添加和删除的次数）
    transient int modCount;//0
    //hash种子数
    transient int hashSeed = 0;
    
    public HashMap() {
        this(DEFAULT_INITIAL_CAPACITY, DEFAULT_LOAD_FACTOR);
    }
    
    //initialCapacity - 16
    //loadFactor - 0.75f
    public HashMap(int initialCapacity, float loadFactor) {
        if (initialCapacity < 0)
            throw new IllegalArgumentException("Illegal initial capacity: " +
                                               initialCapacity);
        if (initialCapacity > MAXIMUM_CAPACITY)
            initialCapacity = MAXIMUM_CAPACITY;
        
        //Float.isNaN(loadFactor)考虑到外界传入的数据：new Float(0.0/0.0)
        if (loadFactor <= 0 || Float.isNaN(loadFactor))//NaN：Not a Number
            throw new IllegalArgumentException("Illegal load factor: " +
                                               loadFactor);

        this.loadFactor = loadFactor;
        threshold = initialCapacity;
        init();
    }
    
    //留给子类去重写的方法
    void init() {
    }
    
    //key - null
    //value -"写代码"
    public V put(K key, V value) {
        //第一次添加元素时进入的判断
        if (table == EMPTY_TABLE) {
            //1.计算阈值 -- 12
            //2.初始化hash数组
            //3.获取了hash种子数
            inflateTable(threshold);
        }
        //判断key是否为null
        if (key == null)
            return putForNullKey(value);
        
        //获取key的hash值
        int hash = hash(key);
        //利用hash值计算在数组中的下标
        int i = indexFor(hash, table.length);
        //一旦进入到该循环，就说明hash碰撞了
        //e - 0x003
        for (Entry<K,V> e = table[i]; e != null; e = e.next) {
            Object k;
            if (e.hash == hash && ((k = e.key) == key || key.equals(k))) {
                //oldValue - 赏月
                V oldValue = e.value;
                e.value = value;
                e.recordAccess(this);
                return oldValue;//返回被替换的值
            }
        }

        modCount++;
        addEntry(hash, key, value, i);
        return null;
    }
    
    //value - 写代码
    private V putForNullKey(V value) {
        //e - 0x004
        for (Entry<K,V> e = table[0]; e != null; e = e.next) {
            if (e.key == null) {
                //oldValue - 打游戏
                V oldValue = e.value;
                e.value = value;
                e.recordAccess(this);
                return oldValue;//返回被替换的值
            }
        }
        modCount++;
        addEntry(0, null, value, 0);
        return null;
    }
    
    //hash - 0
    //key - null
    //value - "打游戏"
    //bucketIndex - 0
    void addEntry(int hash, K key, V value, int bucketIndex) {
        //判断是否扩容
        if ((size >= threshold) && (null != table[bucketIndex])) {
            //进入到扩容的方法
            resize(2 * table.length);
            //重新计算key的hash值
            hash = (null != key) ? hash(key) : 0;
            //重新利用hash值计算在数组中的下标
            bucketIndex = indexFor(hash, table.length);
        }

        createEntry(hash, key, value, bucketIndex);
    }
    
    //newCapacity - 32
    void resize(int newCapacity) {
        Entry[] oldTable = table;
        int oldCapacity = oldTable.length;
        //如果老的容量等于了数组的最大容量,直接将阈值变作为int类型的最大值，那么以后就不会扩容
        if (oldCapacity == MAXIMUM_CAPACITY) {
            threshold = Integer.MAX_VALUE;
            return;
        }

        Entry[] newTable = new Entry[newCapacity];//new Entry[32];
        //将老的数组中的数据复制到新的数组中，并且重新计算hash种子数
        transfer(newTable, initHashSeedAsNeeded(newCapacity));
        //将新数组的地址赋值给hash数组
        table = newTable;
           //重新计算阈值
        threshold = (int)Math.min(newCapacity * loadFactor, MAXIMUM_CAPACITY + 1);
    }
    
    //多线程下可能会出现hash回环的问题
    void transfer(Entry[] newTable, boolean rehash) {
        int newCapacity = newTable.length;
        for (Entry<K,V> e : table) {
            while(null != e) {
                Entry<K,V> next = e.next;
                if (rehash) {
                    e.hash = null == e.key ? 0 : hash(e.key);
                }
                int i = indexFor(e.hash, newCapacity);
                e.next = newTable[i];
                newTable[i] = e;
                e = next;
            }
        }
    }
    
    //hash - 0
    //key - null
    //value - "打游戏"
    //bucketIndex - 0
    void createEntry(int hash, K key, V value, int bucketIndex) {
        //e = null
        Entry<K,V> e = table[bucketIndex];
        //创建新的Entry对象，并把Entry对象的地址赋值到hash数组中当前下标的位置上
        table[bucketIndex] = new Entry<>(hash, key, value, e);//0x004
        size++;
    }
    
    static int indexFor(int h, int length) {
         //h 		- 0000,0000,0000,0000,0000,0000,0001,0100
        //length-1  - 0000,0000,0000,0000,0000,0000,0000,1111
        //			  0000,0000,0000,0000,0000,0000,0000,0100
        return h & (length-1);
    }
    
    final int hash(Object k) {
        int h = hashSeed;
        //判断种子数是否为0 并且 判断key是否为String类型
        if (0 != h && k instanceof String) {
            //获取到有hash种子数参与计算出的hash值
            return sun.misc.Hashing.stringHash32((String) k);
        }

        h ^= k.hashCode();

        h ^= (h >>> 20) ^ (h >>> 12);
        return h ^ (h >>> 7) ^ (h >>> 4);
    }
    
    //toSize - 16
    private void inflateTable(int toSize) {
        // 传入toSize，获取到>=toSize的2的幂的数字
        //capacity - 16
        int capacity = roundUpToPowerOf2(toSize);

        //threshold - 12
        threshold = (int) Math.min(capacity * loadFactor, MAXIMUM_CAPACITY + 1);
        table = new Entry[capacity];
        initHashSeedAsNeeded(capacity);
    }
    
    //根据容量计算出hash种子数
    final boolean initHashSeedAsNeeded(int capacity) {
        boolean currentAltHashing = hashSeed != 0;
        boolean useAltHashing = sun.misc.VM.isBooted() &&
                (capacity >= Holder.ALTERNATIVE_HASHING_THRESHOLD);
        boolean switching = currentAltHashing ^ useAltHashing;
        if (switching) {
            hashSeed = useAltHashing
                ? sun.misc.Hashing.randomHashSeed(this)
                : 0;
        }
        return switching;
    }
    
    //number - 16
    private static int roundUpToPowerOf2(int number) {
        //Integer.highestOneBit(数字)：将数字二进制中最高位的1保留下来，其余全部变为0
        return number >= MAXIMUM_CAPACITY
                ? MAXIMUM_CAPACITY
                : (number > 1) ? Integer.highestOneBit((number - 1) << 1) : 1;
    }
    
    //映射关系类/节点类
    static class Entry<K,V> implements Map.Entry<K,V> {
        final K key; ----- 键
        V value; --------- 值
        Entry<K,V> next; - 下一个节点的内存地址
        int hash; -------- key的hash值

        Entry(int h, K k, V v, Entry<K,V> n) {
            value = v;
            next = n;
            key = k;
            hash = h;
        }
    }
    
}

TreeMap<Student, String> map = new TreeMap<>();

TreeMap<Student, String> map = new TreeMap<>(new Comparator<Student>() {
            //排序规则：判断两个学生是否相同，不相同就按照名字长度排序，名字长度一致就按照年龄排序
            @Override
            public int compare(Student o1, Student o2) {
                if(o1.equals(o2)){
                    return 0;
                }
                
                int nameLen1 = o1.getName().length();
                int nameLen2 = o2.getName().length();
                if(nameLen1 != nameLen2){
                    return nameLen1 - nameLen2;
                }
                
                int age1 = o1.getAge();
                int age2 = o2.getAge();
                if(age1 != age2){
                    return age1-age2;
                }
                return 1;
            }
        });

TreeMap的数据结构是红黑树

红黑树的理解：小的放右边，大的方左边

红黑树的特点：根节点到任意叶子节点的黑树都是一样的

红黑树的优点：查询快
