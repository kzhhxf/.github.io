1.4 反射分析普通方法

public class TestMethod {
    
    public static void main(String[] args) throws Exception {
        Class<?> clazz = Class.forName("com.qf.day26.reflect.User");
        User user = (User) clazz.newInstance();
        //获取指定的方法
        Method m = clazz.getMethod("show");
        //System.out.println(m);
        //执行这个方法--调用这个方法 参数1是对象，参数2就是方法参数
        m.invoke(user);
        
        //调用私有的方法
        Method dm = clazz.getDeclaredMethod("test");
        dm.setAccessible(true);
        dm.invoke(user);
        
        //调用带参带返回的方法
        Method m1 = clazz.getMethod("add",int.class,int.class);
        System.out.println(m1.getModifiers());
        System.out.println(Arrays.toString(m1.getParameters()));
        int res = (int) m1.invoke(user, 6,6);//调用方法并获得返回值
        System.out.println(res);
        
        //获取所有的方法
        System.out.println("获取到User类中所有的方法有：------------------------------");
        Method[] ms = clazz.getMethods();//这个获取到是自己和父类所有的公有构造
        for(Method m0:ms) {
            System.out.println(m0);
        }
        //获取自己类的方法
        System.out.println("获取到User自己类中所有的方法有：------------------------------");
        Method[] dms = clazz.getDeclaredMethods();//这个获取到是自己和父类所有的公有构造
        for(Method m0:dms) {
            System.out.println(m0);
        }
    }

}

1.5 分析属性

public class TestField {
    public static void main(String[] args) throws Exception {
        // 分析属性
        Class<?> clazz = Class.forName("com.qf.day26.reflect.User");
        User user = (User) clazz.newInstance();
        // 公有的属性
        Field field = clazz.getField("age");
        System.out.println(field);
        // 给这个属性赋值
        System.out.println(user);
        field.set(user, 10);
        System.out.println(user);
        // 私有的属性
        Field df = clazz.getDeclaredField("name");
        System.out.println(Modifier.toString(df.getModifiers()));//获取访问修饰符
        df.setAccessible(true);
        System.out.println(field);
        // 给这个属性赋值
        System.out.println(user);
        df.set(user, "张三丰");
        System.out.println(user);
    }
}

练习：
    设计一个工具方法，把一个对象装换为json字符串
    User user  = new User("张三",20);
--->{"name":"张三","age":"20"}
    Student s = new Student("李四",20,99);
--->{"name":"张三","age":"20","score":"99"}
   反射的思想：
   你只要传给我一个对象，我就能够获取到你这个对象的类型---模板，通过反射获取所有的属性和方法，属性值。
public class JsonUtil {
    /*
     *Student s = new Student("李四",20,99);
      {"name":"张三","age":"20","score":"99"}
     */
    public static String getJson(Object obj) throws Exception {
        String str = "{";
        //通过反射获取到所有的属性名和属性值
        Class<?> clazz = obj.getClass();
        //获取所有的属性
        Field[] dfs = clazz.getDeclaredFields();
        for(int i=0;i<dfs.length;i++) {
            Field f = dfs[i];
            f.setAccessible(true);//打开访问权限
            //获取属性名			
            String fieldName = f.getName();
            str += "\""+fieldName+"\":\""+f.get(obj)+"\"";
            if(i!=dfs.length-1) {
                str+=",";
            }
        }
        str += "}";
        return str;
    }
}

public class Test {
    
    public static void main(String[] args) throws Exception {
        //多态---手动切换数据源---坏处
        //缺点：1、要重启应用才能生效
        //      2、交付的程序已经没有java，.class 文件修改不方便
        //      3、即便我交付的是java文件，对方的运维人员不懂java,不知道到哪儿改	
        //解决方案：使用反射+配置文件来改造----反射可以通过字符串来创建对象
        //DataSource ds = new C3P0DataSource();
        //读取配置文件
        Properties prop = new Properties();//读配置文件的一个类
        //获取配置文件的流
        InputStream is = Test.class.getClassLoader().getResourceAsStream("ds.properties");
        //加载文件
        prop.load(is);
        //获取到用户配置的数据源
        String dsProp = prop.getProperty("ds");
        String className = "";
        if(dsProp.equals("druid")) {
            className = "com.qf.day26.ds.DruidDataSource";
        }else if(dsProp.equals("c3p0")) {
            className = "com.qf.day26.ds.C3P0DataSource";
        }else {
            throw new RuntimeException("数据源配置有误");
        }
        //创建数据源		
        Class<?> clazz = Class.forName(className);
        DataSource ds = (DataSource) clazz.newInstance();
        ds.getConnection();
    }
}

1.6 反射操作数组

/*
 * 数组的万能扩展
 */
public class ArrayUtil {	
    /**
     * 
     * @param array  --需要扩容的数组
     * @param length  --新长度
     * @return 扩容后的数组
     */
    public static<T> T copyArray(T array,int length) {
        //思路是先获取到数组的类型
        Class<?> clazz = array.getClass();
        //获取数组中元素的类型
        Class<?> ct = clazz.getComponentType();
        //创建新数组
        T t = (T) Array.newInstance(ct, length);
        //使用循环把array中的元素copy到t这个数组中
        for(int i=0;i<Array.getLength(array);i++) {
            Array.set(t, i, Array.get(array, i));
        }		
        return t;
    }

}

1.7、反射异常总结

1、ClassNotFoundException:找不到类的异常--类的完全限定名写错了
2、NoSuchMethodException:没有这个方法
3、IllegalAccessException：非法访问异常--打破封装
4、IllegalArgumentException：非法参数--我要的参数和你传递的参数不一致
5、ClassCastException:类型转换异常

理解为什么要用注解，注解能够干什么
    注解是程序员和架构师沟通的桥梁-------配置文件也是
    配置----》注解+配置----》注解（去配置文件）
 
思考：
    注解应该由谁定义？
       架构师----设计框架的人
    注解谁来使用？
       程序员
解决两个事：
    定义什么注解，别人怎么用这个注解表达程序员的想法
       1、我们定义的注解能在哪里写？
            元注解----在注解上面写的注解
            @Target({}):指定注解能够使用的位置
       2、写什么？
          注解中通过属性定义该写什么
@Target(ElementType.TYPE)
public @interface Ds {	
    String className();//这个字符串属性规定了你必须配置数据源的路径
}
       3、指定注解存活的时间---默认是类文件
        @Retention(value=RetentionPolicy.RUNTIME)           
            SOURCE:只保留到源文件
            CLASS：保留到字节码文件
            RUNTIME：保留到运行时
注意：如果属性名是value(),则可以省略这个名字，如果不是，这个名字就不能省略
